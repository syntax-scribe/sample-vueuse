### `arrayEquals(a: T[], b: T[]): boolean`

<details><summary>Code</summary>

```ts
function arrayEquals<T>(a: T[], b: T[]): boolean {
  if (a.length !== b.length)
    return false

  for (let i = 0; i < a.length; i++) {
    if (a[i] !== b[i])
      return false
  }
  return true
}
```
